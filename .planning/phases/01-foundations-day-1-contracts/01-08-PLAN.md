Key idioms:
- `@Container static` (NOT instance field) → singleton container reuse across all tests in the class (Pitfall #22 mitigation; RESEARCH §4.7 line 666: "use `@Container` static field for singleton container reuse").
- `withCopyFileToContainer` (NOT `withInitScript`) → routes through Postgres' canonical `/docker-entrypoint-initdb.d/` mechanism which interpolates env vars in `.sh` scripts (RESEARCH §4.4 line 500).
- `DockerImageName.parse(...).asCompatibleSubstituteFor("postgres")` → required because `pgvector/pgvector:pg16` is a postgres derivative; without this, Testcontainers' image-name validation rejects it.
- `withDatabaseName("n11") + withUsername("postgres") + withPassword("postgres-test")` → values populate `POSTGRES_DB`, `POSTGRES_USER`, `POSTGRES_PASSWORD` env vars in the container — these are what init.sh's `psql --username "$POSTGRES_USER" --dbname "$POSTGRES_DB"` references.

```kotlin
// infra-tests/build.gradle.kts
//
// Test-only Gradle module. Hosts the cross-schema boundary smoke test (D-05) that
// verifies Plan 01-03's `infra/postgres/init.sh` enforces ARCH-09 at runtime.
//
// Plugin discipline (Cross-Cutting #1, RESEARCH §4.11 line 1183):
//   - `java` plugin ONLY. NO `org.springframework.boot` plugin — this module never
//     boots Spring, so a Boot plugin would synthesize a fake main class and fail
//     the executable-jar build. The root `subprojects {}` block already applies
//     `java` and `io.spring.dependency-management`, so this file just adds deps.
//
// Why no `src/main`: pure test module. Hence we explicitly skip the `compileJava`
// task by leaving `src/main/java` absent. Gradle handles this gracefully —
// `compileJava` no-ops when the source set is empty.
//
// init.sh classpath strategy:
//   The single source of truth for init.sh lives at `infra/postgres/init.sh` (root
//   project, written by Plan 01-03). The `copyInitScript` task below copies it into
//   `src/test/resources/init.sh` before `processTestResources` runs, so Testcontainers'
//   `MountableFile.forClasspathResource("init.sh")` resolves it from the test JAR.
//   This prevents drift between production init.sh and a hand-edited test fixture.

plugins {
    // No plugin block needed — root `subprojects { apply(plugin = "java") ... }` covers it.
    // We could redeclare `java` here for clarity but Gradle treats the inherited
    // application as authoritative.
}

dependencies {
    // Testcontainers — versions managed by the testcontainers-bom 2.0.5 imported in
    // the root build's dependency-management block (Plan 01-01).
    testImplementation("org.testcontainers:postgresql")
    testImplementation("org.testcontainers:junit-jupiter")

    // Postgres JDBC driver — needed by Testcontainers' PostgreSQLContainer to open
    // the JdbcUrl returned by getJdbcUrl(), and by our DriverManager.getConnection
    // calls in CrossSchemaDenyTest. Version managed by Spring Boot BOM (root build).
    testImplementation("org.postgresql:postgresql")

    // JUnit 5 — version managed by Spring Boot BOM (`junit-jupiter` 5.10.x via Boot 3.5.14).
    testImplementation("org.junit.jupiter:junit-jupiter")
    testRuntimeOnly("org.junit.platform:junit-platform-launcher")

    // SLF4J — Testcontainers logs container start/stop on slf4j; without a binding,
    // Testcontainers prints a warning but still works. `slf4j-simple` keeps test
    // output readable without dragging Logback into a test-only module.
    testRuntimeOnly("org.slf4j:slf4j-simple")
}

// Copy init.sh from the root project into this module's test resources so the
// Testcontainers test loads it via classpath. Hooked before processTestResources
// so the file is present whenever ./gradlew :infra-tests:test or :build runs.
tasks.register<Copy>("copyInitScript") {
    description = "Copies infra/postgres/init.sh into the test classpath so " +
                  "MountableFile.forClasspathResource(\"init.sh\") resolves it."
    group = "build"
    from(rootProject.layout.projectDirectory.file("infra/postgres/init.sh"))
    into(layout.projectDirectory.dir("src/test/resources"))
    // Fail fast if init.sh is missing (e.g. Plan 01-03 hasn't run yet on a fresh
    // checkout). Better than letting Testcontainers boot and the test fail with
    // a confusing "init.sh not found on classpath" message.
    doFirst {
        val src = rootProject.layout.projectDirectory.file("infra/postgres/init.sh").asFile
        if (!src.exists()) {
            throw GradleException(
                "infra/postgres/init.sh is missing. Run Plan 01-03 first " +
                "(it writes the init script that this test verifies)."
            )
        }
    }
}

tasks.named("processTestResources") {
    dependsOn("copyInitScript")
}

tasks.withType<Test> {
    useJUnitPlatform()
    // Surface Testcontainers logs at INFO so container boot timings appear in CI output —
    // helps diagnose Pitfall #22 if singleton reuse ever regresses.
    systemProperty("org.slf4j.simpleLogger.defaultLogLevel", "info")
    // Long timeout because cold-boot can be slow on first pull of pgvector/pgvector:pg16.
    timeout.set(java.time.Duration.ofMinutes(3))
}
```
