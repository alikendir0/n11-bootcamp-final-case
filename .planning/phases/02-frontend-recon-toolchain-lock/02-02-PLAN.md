Spec template shape (PATTERNS.md per-file homepage.spec.ts contract — clone-and-modify for the 6 siblings):
```typescript
import { test, expect } from '@playwright/test';
import * as fs from 'node:fs';
import * as path from 'node:path';
import { dismissBanners } from '../lib/dismiss-banners';
import { harvestCopy } from '../lib/harvest-copy';
import { harvestColors } from '../lib/harvest-colors';

const SLUG = 'homepage';                              // <-- per-spec
const URL = 'https://www.n11.com/';                   // <-- per-spec
const ELEMENT_ZOOM_SELECTOR = 'header';               // <-- per-spec, optional

const SCREENSHOT_DIR = path.resolve(__dirname, '../../../.planning/intel/screenshots');
const OUTPUT_DIR = path.resolve(__dirname, '../output');

test.describe(`n11 ${SLUG} recon`, () => {
  test.beforeAll(() => {
    fs.mkdirSync(SCREENSHOT_DIR, { recursive: true });
    fs.mkdirSync(OUTPUT_DIR, { recursive: true });
  });

  test(`capture ${SLUG}`, async ({ page }) => {
    const response = await page.goto(URL, { waitUntil: 'networkidle' });
    expect(response?.status()).toBeLessThan(400);     // S-2: hard-fail on 4xx
    await page.waitForTimeout(2500);                   // S-4: gentle rate
    await dismissBanners(page);                        // Pattern 3
    await page.waitForTimeout(500);                    // settle after dismiss

    await page.screenshot({
      path: path.join(SCREENSHOT_DIR, `${SLUG}-fullpage.png`),
      fullPage: true,
    });

    if (ELEMENT_ZOOM_SELECTOR) {
      const el = page.locator(ELEMENT_ZOOM_SELECTOR).first();
      if (await el.count() > 0) {
        await el.screenshot({
          path: path.join(SCREENSHOT_DIR, `${SLUG}-${ELEMENT_ZOOM_SELECTOR.replace(/[^a-z0-9]/gi, '')}-element.png`),
        });
      }
    }

    const phrases = await harvestCopy(page);
    fs.writeFileSync(
      path.join(OUTPUT_DIR, `${SLUG}-phrases.json`),
      JSON.stringify({ page: SLUG, url: page.url(), phrases }, null, 2),
      'utf-8',
    );

    const tokens = await harvestColors(page);
    fs.writeFileSync(
      path.join(OUTPUT_DIR, `${SLUG}-tokens.json`),
      JSON.stringify({ page: SLUG, url: page.url(), tokens }, null, 2),
      'utf-8',
    );

    expect(phrases.length).toBeGreaterThan(5);
  });
});
```
