1. **Create `frontend/scripts/check-frontend-invariants.sh`** — the canonical grep gate. Keep checks tight enough to catch real violations but not so tight they false-positive on incidental matches. The script must be executable.

   ```bash
   #!/usr/bin/env bash
   # Frontend phase 10 invariants — single CI gate enforcing the 11 cross-cutting truths.
   # Run from repo root: bash frontend/scripts/check-frontend-invariants.sh

   set -u
   FAIL=0
   FRONTEND_SRC="frontend/src"

   step() { echo "==> $1"; }
   pass() { echo "    OK"; }
   fail() { echo "    FAIL: $1"; FAIL=$((FAIL + 1)); }

   if [ ! -d "$FRONTEND_SRC" ]; then
     echo "frontend/src/ not found — run from repo root."
     exit 2
   fi

   step "1. All API calls go through VITE_API_BASE_URL (no hardcoded service URLs in src/)"
   # Allow only inside vite-env.d.ts (type declaration). Disallow anywhere else.
   bad=$(grep -rln "localhost:[0-9]" "$FRONTEND_SRC" | grep -v "vite-env.d.ts" || true)
   if [ -n "$bad" ]; then fail "hardcoded localhost in: $bad"; else pass; fi

   step "2. n11_auth_token localStorage key only used in tokenStore.ts"
   bad=$(grep -rln "n11_auth_token" "$FRONTEND_SRC" | grep -v "tokenStore" || true)
   if [ -n "$bad" ]; then fail "n11_auth_token referenced outside tokenStore: $bad"; else pass; fi

   step "3. No dangerouslySetInnerHTML in source"
   bad=$(grep -rln "dangerouslySetInnerHTML" "$FRONTEND_SRC" || true)
   if [ -n "$bad" ]; then fail "dangerouslySetInnerHTML found in: $bad"; else pass; fi

   step "4. Intl.NumberFormat only in lib/format.ts (and its test)"
   bad=$(grep -rln "Intl\\.NumberFormat" "$FRONTEND_SRC" | grep -vE "lib/format\\.(ts|test\\.ts)$" || true)
   if [ -n "$bad" ]; then fail "Intl.NumberFormat used outside lib/format.ts: $bad"; else pass; fi

   step "5. Intl.DateTimeFormat only in lib/format.ts (and its test)"
   bad=$(grep -rln "Intl\\.DateTimeFormat" "$FRONTEND_SRC" | grep -vE "lib/format\\.(ts|test\\.ts)$" || true)
   if [ -n "$bad" ]; then fail "Intl.DateTimeFormat used outside lib/format.ts: $bad"; else pass; fi

   step "6. Route constants centralized in lib/routes.ts (no string literals outside)"
   # Heuristic: literal '/sepetim' or '/giris-yap' should appear ONLY in routes.ts and router.tsx (where literal strings are necessary)
   bad=$(grep -rln "'/sepetim'\\|\"/sepetim\"" "$FRONTEND_SRC" | grep -vE "lib/routes\\.ts$|router\\.tsx$" || true)
   if [ -n "$bad" ]; then fail "Hardcoded '/sepetim' route literal outside routes.ts/router.tsx: $bad"; else pass; fi

   step "7. CTA primary color #1C1C1E in @theme block (recon-locked — NOT n11 orange)"
   if ! grep -q "color-cta-primary-bg: #1C1C1E" "$FRONTEND_SRC/index.css"; then
     fail "@theme block missing --color-cta-primary-bg: #1C1C1E"
   else
     pass
   fi
   bad=$(grep -E "ff6600|ff7a00|f5a623" "$FRONTEND_SRC/index.css" || true)
   if [ -n "$bad" ]; then fail "n11-orange detected on CTA — must be #1C1C1E"; fi

   step "8. RequireAuth guard wraps the auth-required routes in router.tsx"
   if ! grep -q "RequireAuth" "$FRONTEND_SRC/router.tsx"; then
     fail "RequireAuth not used in router.tsx"
   else
     pass
   fi

   step "9. Iyzico return path is /odeme/sonuc — checkout flow uses ROUTES.CHECKOUT_RESULT"
   if ! grep -q "/odeme/sonuc" "$FRONTEND_SRC/lib/routes.ts"; then
     fail "ROUTES.CHECKOUT_RESULT path '/odeme/sonuc' missing"
   else
     pass
   fi

   step "10. Order POST sends Idempotency-Key header (Phase 5 D-05 contract)"
   if ! grep -q "'Idempotency-Key'" "$FRONTEND_SRC/api/orderApi.ts"; then
     fail "Idempotency-Key header not sent on POST /orders — violates Phase 5 D-05"
   else
     pass
   fi

   step "11. No chat-bubble placeholder (Phase 11 territory)"
   bad=$(grep -rinE "chat[- ]?bubble|coming soon|chat ?placeholder|asistan|yapay zeka" "$FRONTEND_SRC" --include='*.ts' --include='*.tsx' || true)
   if [ -n "$bad" ]; then fail "Chat-bubble placeholder detected: $bad"; else pass; fi

   echo
   if [ "$FAIL" -eq 0 ]; then
     echo "All 11 frontend invariants OK."
     exit 0
   else
     echo "$FAIL invariant(s) violated."
     exit 1
   fi
   ```

   **Make it executable:**
   ```bash

   ```

   import { test, expect } from '@playwright/test';
   import { SEED } from './fixtures/seed';

   test('demo flow: search → PDP → add → login → checkout → Iyzico hand-off', async ({ page }) => {
     // 1. Land on homepage
     await page.goto('/');
     await expect(page).toHaveTitle(/n11/i);

     // 2. Search for the seed product
     const searchInput = page.getByPlaceholder('Aradığınız ürün, kategori veya markayı yazınız');
     await searchInput.fill(SEED.query);
     await searchInput.press('Enter');
     await expect(page).toHaveURL(new RegExp(`/arama\\?q=${SEED.query}`));

     // 3. Click first result — locator strategy: first <a> inside the listing grid
     // ProductCard wraps the whole tile in <Link>; pick the first one inside the main listing region
     const firstResult = page.locator('main a[href^="/urun/"]').first();
     await expect(firstResult).toBeVisible({ timeout: 15_000 });
     await firstResult.click();
     await expect(page).toHaveURL(/\/urun\//);

     // 4. Click Sepete Ekle as anonymous → redirected to login
     await page.getByRole('button', { name: 'Sepete Ekle' }).click();
     await expect(page).toHaveURL(/\/giris-yap\?redirectUrl=/);

     // 5. Log in
     await page.getByLabel('E-posta').fill(SEED.email);
     await page.getByLabel('Şifre').fill(SEED.password);
     await page.getByRole('button', { name: 'Giriş Yap' }).click();

     // 6. Arrive back on the PDP (redirectUrl honored)
     await expect(page).toHaveURL(/\/urun\//, { timeout: 10_000 });

     // 7. Add to cart (now authenticated)
     await page.getByRole('button', { name: 'Sepete Ekle' }).click();

     // 8. Toast confirms + header badge increments
     await expect(page.getByText('Ürün sepete eklendi.')).toBeVisible({ timeout: 8_000 });
     await expect(page.getByLabel(/Sepetim/)).toBeVisible();

     // 9. Navigate to cart
     await page.getByRole('link', { name: /Sepetim/ }).first().click();
     await expect(page).toHaveURL(/\/sepetim$/);

     // 10. Click Siparişi Tamamla
     await page.getByRole('button', { name: 'Siparişi Tamamla' }).click();
     await expect(page).toHaveURL(/\/odeme\/adres/);

     // 11. Address step — pick first address (auto-selected by default)
     // If user has no addresses the form expands; ensure they have at least one
     const newAddressBtn = page.getByRole('button', { name: 'Adresi Kaydet' });
     if (await newAddressBtn.isVisible({ timeout: 1_000 }).catch(() => false)) {
       await page.getByLabel('Adres Başlığı').fill('Ev');

       await page.getByLabel('Telefon').fill('05551234567');
